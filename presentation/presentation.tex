\begin{frame}{Object Information}
  \begin{itemize}
    \item Problem: Es gibt in OSM Gebäude, denen keine Straße zugewiesen wurde
    \item Ziel: Möglichst passende Straße zuweisen
    \item Lösung:
    \begin{itemize}
      \item Balancierter KD-Tree mit allen Straßenpunkten: $\mathcal{O}(n \log(n))$
      \item Für alle Gebäude ohne zugewiesene Straße:
      \begin{itemize}
        \item Finde nächsten Straßenpunkt mit KD-Tree
        \item Weise Gebäude diese Straße zu
      \end{itemize}
    \end{itemize}
  \end{itemize}
