\begin{frame}{Substring-Suche (Suffix Array)}
  \begin{itemize}
    \item Inverted Index bauen
    \begin{itemize}
      \item Volle Gebäudeadresse umwandeln (ß $\to$ ss, ä $\to$ ae, Trennen bei Leerzeichen, ...)
      \item Dictionary: \{$<$Adresse[...]$>$ : \{$<$Gebäude-ID$>$, ...\}\}
    \end{itemize}
    \item Suffix Array aus \textit{InvertedIndex.keys()}
    \begin{itemize}
      \item Effiziente Konstuktion mit Rängen: $\mathcal{O}(n \log^2(n))$
      \begin{itemize}
        \item Sortieren nach $2^0 = 1$ Zeichen
        \item Sortieren nach $2^1 = 2$ Zeichen
        \item ...
        \item Sortieren nach $2^i$ Zeichen
      \end{itemize}
      \item Hinzukommenden Zeichen der nächsten Iteration $2^{i + 1}$ sind Präfix eines anderen Suffixes
      \item "Nur noch" Vergleich zweier Zahlen (ranks)
    \end{itemize}
    \item Query:
    \begin{itemize}
      \item Querytext umwandeln
      \item $\forall q \in query: buildings = search(q) \cap_{galloping} buildings$
    \end{itemize}
  \end{itemize}
\end{frame}


\begin{frame}{Heuristik}
  \begin{itemize}
    \item Rangbasiert
    \item \textit{rank = 0 \\
    if (inViewBox()) rank += 200 \\
    rank += $150 \cdot \dfrac{1}{1 + \textit{distanceToCenter()}}$ \\
    rank += $180 \cdot \textit{queryMatch()}$}
  \end{itemize}

\begin{frame}{Closest-to-Query}
  \begin{itemize}
    \item Syntax: \textit{Closest $\left<X\right>$ to $\left<Y\right>$}
    \item $\widehat{Y} = sort(query(\left<Y\right>))[0]$
    \item $\widehat{X} = min_{dist(\cdot, \widehat{Y})}\{query(\left<X\right>)\}$
  \end{itemize}

\end{frame}
